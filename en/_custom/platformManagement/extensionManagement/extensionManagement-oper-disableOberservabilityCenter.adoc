  monitoring:
    whizard:
      enabled: false          # Set the whizard:enabled parameter to false

status:
  alerting:
    enabledTime: XXXX-XX-XXXXX:XX:XXXXX
    status: enabled           # Delete alerting:status:enabled
  monitoring:
    enabledTime: XXXX-XX-XXXXX:XX:XXXXX
    status: enabled           # Delete monitoring:status:enabled
